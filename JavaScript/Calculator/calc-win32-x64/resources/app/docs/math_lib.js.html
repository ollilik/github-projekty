/**
 * @brief Rounding number to true value
 * 
 * @param number number
 * @return rounded number
 */
function strip(number) 
{
    number = parseFloat(number).toPrecision(15)
    number = Math.round(number * 1000000000000000) / 1000000000000000
    return number
}

/**
 * @brief Method Adding
 * 
 * @param a first number
 * @param b number that will be added to first number
 * @return result of method
 */
function add(a,b)
{
    result = a + b
    return strip(result)
}

/**
 * @brief Method Subtraction
 * 
 * @param a first number
 * @param b number that will be subtracted from first number
 * @return result of method
 */
function sub(a,b)
{
    result = a - b
    return strip(result)
}

/**
 * @brief Method Multiply

function mul(a,b)
{
    result = a * b
    return strip(result)

/**
 * @brief Method Division
 * 
 * @param a divident
 * @param b divisor
 * @return result of method
 */
function div(a,b)
{
    if(b === 0)
        throw new Exception("Divisor can not be 0")
    result = a / b
    return strip(result)
}

/**
 * @brief Method Factorial
 * 
 * @param a number to be factorised
 * @return result of method
 */
function fact(a)
{
    result = 1
    if(a < 0)
        throw new Exception("factorised number must be higher or equal than 0")
    if(Math.round(a) != a)
        throw new Exception("number must be int")
    var i = 2
    for(i; i < a + 1; i++)
    {
        result = result * i
    }
    return strip(result)
}

/**
 * @brief Method Pow
 * 
 * @param a base
 * @param b power
 * @return result of method
 */
function pow(a,b)
{
    result = Math.pow(a,b)
    return strip(result)
}

/**
 * @brief Method Logarithm
 * 
 * @param a base
 * @param b number
 * @return result of method
 */
function log(a,b)
{
    if(a < 0)
        throw new Exception("number must be higher than 0")
    if(b < 0)
        throw new Exception("base must be higher than 0")
    result = Math.log(b) / Math.log(a)
    return strip(result)
}

/**
 * @brief Method Sqrt
 * 
 * @param a number
 * @param b base
 * @return result of method
 */
function sqrt(a,b)
{
    if(a < 0 || b < 0)
        throw new Exception("base and number must be higher than 0")
    result = pow(a,1/b)
    return strip(result)
}
